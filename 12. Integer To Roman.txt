class Solution {
    public String intToRoman(int num) {
       //   Define Roman numerals and their values in descending order
    int [] values = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
    String [] symbols = {"M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"};

    StringBuilder result = new StringBuilder(); //the string we are going to be returning

    for (int i=0; i<values.length;i++){ //loops to through the values
        while (num >= values[i]){ //this loop subtracts the value from num until its less than it to move on to the next
            num = num - values[i]; 
            result.append(symbols[i]); //then it adds the symbol to result
        }
    }
    return result.toString();
//I matched the symbols with the order of values 
        
    }
}
