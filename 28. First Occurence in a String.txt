class Solution {
    public int strStr(String haystack, String needle) { 
        for (int i = 0, j = needle.length(); j<=haystack.length(); i++,j++){//loop two iterations one is needle length and other is base
            if (haystack.substring(i,j).equals(needle)){// if haystack has a substring from i to the length of needle then return i
                return i;
            }
        }
        return -1; //if not return -1
        
    }
}
