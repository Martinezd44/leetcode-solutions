class Solution(object):
    def isPalindrome(self, x):
        if x<0 or (x % 10 ==0 and x!=0):
            return False #Checks for cases

        reversed = 0
        while x > reversed: # loop takes reversed half of the digits then first two of x
                reversed = reversed * 10 +x %10
                x//=10

        return x == reversed or x== reversed//10  ## then it compares it to see if its a palinedrome

        :type x: int
